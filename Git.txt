git add -A
	--all   - Добавляет файлы, папки и вложенные элементы
	main.py - Добавляет сам файл
	.       - Добавляет всё файлы из текущей директории(без вложенных элементов)

git branch - просмотр веток
git branch test - создание новой ветки
git branch -D test - удаление ветки
git checkout "имя ветки" - переход на другую ветку

git checkout -b "new" - создание и переход на новую ветку с именем "new"
